} function normType(t) { if (!t) return null; const s = String(t).toLowerCase().replace(/[\s_-]+/g, ""); if (["delivered","delivery","send","sent"].some(x => s.includes(x))) return "sent"; if (["open","opened"].some(x => s.includes(x))) return "opened"; if (["click","clicked"].some(x => s.includes(x))) return "clicked"; if (["bounce","bounced","hardbounce","softbounce"].some(x => s.includes(x))) return "bounced"; if (["complaint","spam"].some(x => s.includes(x))) return "spam"; if (["unsub","unsubscribe","unsubscribed"].some(x => s.includes(x))) return "unsubscribed"; if (["drop","dropped","skip","skipped"].some(x => s.includes(x))) return "dropped"; if (["defer","deferred"].some(x => s.includes(x))) return "deferred"; return String(t).toLowerCase(); } // Server-side fallback for GHL 4-col export rows (if UI ever changes) function mapServerSideRowFallback(row) { // If row has the 4 GHL headers exactly, remap const keys = Object.keys(row).map(k => k.toLowerCase().trim()); const type = normType(status); const ts = updated ? new Date(updated).toISOString() : new Date().toISOString(); return { email, event_type: type || "sent", first_name: first || null, last_name: last || null, event_timestamp: ts };

// GTR Core - Campaign Manager Worker (Supabase version) const CORS = { "Access-Control-Allow-Origin": "*", "Access-Control-Allow-Methods": "GET, POST, OPTIONS", "Access-Control-Allow-Headers": "Content-Type, X-Admin-Secret", }; const json = (data, status = 200) => new Response(JSON.stringify(data), { status, headers: { ...CORS, "Content-Type": "application/json" }, }); const err = (message, status = 400, detail) => json({ error: message, detail }, status); function needEnv(env) { const miss = []; if (!env.SUPABASE_URL?.startsWith("https://")) miss.push("SUPABASE_URL"); if (!env.SUPABASE_SERVICE_ROLE) miss.push("SUPABASE_SERVICE_ROLE"); if (!env.ADMIN_SECRET) miss.push("ADMIN_SECRET"); return miss; } async function sfetch(env, pathWithQuery, init = {}) { const url = `${env.SUPABASE_URL}/rest/v1/${pathWithQuery}`; return fetch(url, { ...init, headers: { "apikey": env.SUPABASE_SERVICE_ROLE, "Authorization": `Bearer ${env.SUPABASE_SERVICE_ROLE}`, ...(init.headers || {}), }, }); } export default { async fetch(req, env) { try { if (req.method === "OPTIONS") return new Response(null, { status: 204, headers: CORS }); // env check const miss = needEnv(env); if (miss.length) return err(`Missing env: ${miss.join(", ")}`, 500); // auth const admin = req.headers.get("X-Admin-Secret"); if (admin !== env.ADMIN_SECRET) return err("Unauthorized", 401); const url = new URL(req.url); const { pathname, searchParams } = url; // ---- ROUTES ---- // Health if (req.method === "GET" && pathname === "/health") { return json({ ok: true, ts: new Date().toISOString() }); } // Clients list -> { rows: [...] } if (req.method === "GET" && pathname === "/clients") { const r = await sfetch( env, `clients?select=location_id,business_name,full_name,email&order=business_name.asc,full_name.asc&limit=1000`, { method: "GET" } ); if (!r.ok) return err("clients fetch failed", r.status, await r.text()); const rows = await r.json(); return json({ rows }); } // Get campaigns for a client (location_id) -> { rows: [...] } if (req.method === "GET" && pathname === "/campaigns") { const location_id = searchParams.get("location_id") || ""; if (!location_id) return err("location_id required"); const r = await sfetch( env, `campaign_campaigns?select=campaign_id,campaign_name,description,notes&location_id=eq.${encodeURIComponent(location_id)}&order=campaign_id.desc`, { method: "GET" } ); if (!r.ok) return err("campaigns fetch failed", r.status, await r.text()); const rows = await r.json(); return json({ rows }); } // Add campaign -> { campaign_id } if (req.method === "POST" && pathname === "/campaigns") { const ct = (req.headers.get("content-type") || "").toLowerCase(); if (!ct.includes("application/json")) return err("Expected application/json", 415); const body = await req.json(); const location_id = String(body.location_id || "").trim(); const campaign_name = String(body.campaign_name || "").trim(); "").toString(); const notes = (body.notes ?? "").toString(); if (!campaign_id) return err("campaign_id required"); if (!run_label) return err("run_label required"); const row = { campaign_id, run_label, subject, notes, source: "admin_ui", is_backfill: true, loaded_at: new Date().toISOString(), updated_at: new Date().toISOString(), }; const r = await sfetch(env, `campaign_runs`, { method: "POST", headers: { "Content-Type": "application/json", "Prefer": "return=representation", }, body: JSON.stringify([row]), }); const text = await r.text(); if (!r.ok) return err("run insert failed", r.status, text); let rows = []; try { rows = JSON.parse(text); } catch {} const run_id = rows?.[0]?.run_id; if (!run_id) return err("run insert returned no id", 500, text); return json({ run_id }); } return err("Not Found", 404); } catch (e) { return err("Unhandled error", 500, String(e?.message || e)); } }, };
